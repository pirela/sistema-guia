# ============================================
# SUPABASE - Base de Datos
# ============================================
# Obtén estos valores en: https://supabase.com/dashboard/project/_/settings/api
#
# Project URL (ejemplo: https://abcdefghijklm.supabase.co)
NEXT_PUBLIC_SUPABASE_URL=https://tu-proyecto-id.supabase.co

# Anon/Public Key (empieza con eyJ...)

# ============================================
# SHOPIFY - Integración con Tienda
# ============================================
# Obtén estos valores creando una Custom App en tu tienda Shopify
# Instrucciones: Settings → Apps → Develop apps → Create app
#

# URL de tu tienda (ejemplo: mi-tienda.myshopify.com)
# ⚠️ SIN https:// al inicio
NEXT_PUBLIC_SHOPIFY_SHOP_URL=tu-tienda.myshopify.com

# --- Shopify Dev Dashboard (recomendado): renovación automática del token (~24h en Shopify)
# Si defines AMBOS, el proyecto pide un token nuevo antes de que caduque (no uses SHOPIFY_ACCESS_TOKEN)
SHOPIFY_CLIENT_ID=tu-client-id
SHOPIFY_CLIENT_SECRET=shpss_tu_secreto

# --- Opcional: token estático (app personalizada heredada con shpat_ fijo)
# Si no usas CLIENT_ID/SECRET, descomenta y rellena:
